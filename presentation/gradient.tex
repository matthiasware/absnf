	\begin{flalign*}
		\gamma, \Gamma
	\end{flalign*}
	\begin{flalign*}
		\gamma &= b + Y \Sigma(I-L\Sigma)^{-1} a \\
		\Gamma &= J + Y \Sigma(I-L\Sigma)^{-1} Z
	\end{flalign*}
	\begin{flalign*}
		\Sigma = Diag(Sign(\Delta z))
	\end{flalign*}

	\begin{pmatrix}
	\Delta z \\
	\Delta y
	\end{pmatrix}

	\begin{pmatrix}
	a \\
	b
	\end{pmatrix}

	Z & L \\
	J & Y 
	\end{pmatrix}
	\times
	\begin{pmatrix}
	\Delta x \\
	|\Delta z |
	\end{pmatrix}
	\end{flalign*}
